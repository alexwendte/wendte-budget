import axios from 'axios'
 
let api: any
let isLoggedIn: boolean
const envBaseURL = process.env.REACT_APP_API_URL
 
const getData = (res: { data: JSON }) => res.data

interface IRequests {
  delete: (url: string) => Promise<any>
  get: (url: string) => Promise<any>
  post: (url: string, body: object) => Promise<any>
  put: (url: string, body: object) => Promise<any>
}
 
const requests: IRequests = {
  delete: (url: string) => api.delete(url).then(getData),
  get: (url: string) => api.get(url).then(getData),
  post: (url: string, body: object) => api.post(url, body).then(getData),
  put: (url: string, body: object) => api.put(url, body).then(getData),
}

interface IUser {
  token: string
}

const auth = {
  login: (form: object) =>
    requests.post('/auth/login', form).then((data: { user: IUser }) => {
      login({ token: data.user.token })
      return data
    }),
  logout: () => {
    logout()
    return Promise.resolve({ user: null })
  },
  me() {
    if (!isLoggedIn) {
      return Promise.resolve({ user: null })
    }
    return requests.get('/auth/me').catch((err: any) => {
      if (err.response.status === 401) {
        logout()
        // return { user: null }
      }
      return Promise.reject(err)
    })
  },
  register: (form: any) =>
    requests.post('/auth/register', form).then(data => {
      login({ token: data.user.token })
      return data
    }),
}

const users = {
  create: (user: object) => requests.post('/users', user),
  delete: (id: string) => requests.delete(`/users/${id}`),
  get: (id?: string) => requests.get(id ? `/users/${id}` : '/users'),
  update: (id: string, updates: any) => requests.put(`/users/${id}`, updates),
}

const transactions = {
  create: (transaction: object) => requests.post('/transactions', transaction),
  delete: (id: string) => requests.delete(`/transactions/${id}`),
  get: (id?: string) => requests.get(id ? `/transactions/${id}` : '/transactions'),
  update: (id: string, updates: any) => requests.put(`/transactions/${id}`, updates),
}

// In the server api can use req.user.id to get the categories associated with a user
const categories = {
  create: (post: object) => requests.post('/categories', post),
  delete: (id: string) => requests.delete(`/categories/${id}`),
  get: (id?: string) => requests.get(id ? `/categories/${id}` : '/categories'),
  update: (id: string, updates: any) => requests.put(`/categories/${id}`, updates),
}
 
function logout() {
  window.localStorage.removeItem('token')
  init({ token: null })
}
function login({ token }: { token: string }) {
  window.localStorage.setItem('token', token)
  init({ token })
}
 
function init({
  token = window.localStorage.getItem('token'),
  baseURL = (api && api.defaults.baseURL) || envBaseURL,
  axiosOptions = { headers: {} },
} = {}) {
  isLoggedIn = Boolean(token)
  api = axios.create({
    baseURL,
    ...axiosOptions,
    headers: {
      authorization: token ? `Bearer ${token}` : undefined,
      ...axiosOptions.headers,
    },
  })
}
 
const restApi = {
  auth,
  categories,
  init,
  transactions,
  users,
}
 
export default restApi
