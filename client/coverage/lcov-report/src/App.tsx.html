import User from 'components/User'
import * as React from 'react'
import styled, { ThemeProvider } from 'styled-components'
import { transactions as fakeTransactions } from 'utils/fakeData'
import { ITransaction } from 'utils/types'
import ItemForm from './ItemForm'
import './styles/App.css'
 
import TransactionDisplay from './transactionDisplay/TransactionDisplay'
 
const theme = {
  black: '#222222',
  green: '#0ac775',
  grey: '#354041',
  lightGrey: '#f1f3f4',
  maxWidth: '1000px',
  primary: '#66B9Bf',
  white: '#fff',
}
 
interface IState {
  items: ITransaction[]
}

class App extends React.Component<{}, IState> {
  public state: IState = {
    items: fakeTransactions,
  }
 
  // I can just pass user to ItemForm, no need for context!

  public render() {
    const { items } = this.state
    return (
      <ThemeProvider theme={theme}>
        <AppWrapper className="hi" data-testid="app">
          <User>
            {({ user }: { user: object }) => (
              <>
                <ItemForm user={user} />
                <TransactionDisplay items={items} />
              </>
            )}
          </User>
        </AppWrapper>
      </ThemeProvider>
    )
  }
}
 
export default App
 
const AppWrapper = styled.div`
  display: flex;
  flex-direction: column;
  align-items: center;
  padding-bottom: 4rem;
`
