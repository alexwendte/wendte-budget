import * as React from 'react'
import styled from 'styled-components'
import restApi from 'utils/api'
import { elevation, media } from 'utils/mixins'
import { ITransaction } from 'utils/types'
import TransactionRow from './TransactionRow'
 
interface IState {
  readOnly: boolean
  transactions?: ITransaction[]
  loading: boolean
}
 
class TransactionDisplay extends React.Component<{}, IState> {
  state = {
    loading: true,
    readOnly: true,
    transactions: undefined,
  }
 
  componentDidMount() {
    restApi.transactions
      .get()
      .then(
        ({ transactions }) => this.setState({ transactions, loading: false }),
        (error: Error) => console.error(error)
      )
  }
 
  editTransaction = () => {
    this.setState(({ readOnly }) => ({ readOnly: !readOnly }))
  }

  render() {
    const { readOnly, transactions, loading } = this.state
    // I imagine the save changes will be a button with type submit.
    // Doesn't that mean that the whole table is a form?
    // How will I decide what data was changed? I only want to update the transaciton that was changes.
    // Each transaction should have an id. When it is changed, I will include that id in the list of id's that were changed.
    return loading ? (
      <div data-testid="loading" hidden={true} />
    ) : (
      <TransactionTable>
        <h2 className="heading">Recent Transactions</h2>
        <Controls>
          <Edit onClick={this.editTransaction}>Edit Transactions</Edit>
          <Save>Save Changes</Save>
        </Controls>
        <TableItems>
          <TableHeader>
            <div>Title</div>
            <div className="date">Date</div>
            <div className="category">Category</div>
            <div className="amount">Amount</div>
          </TableHeader>
          {transactions &&
            (transactions as ITransaction[]).map((transaction: ITransaction) => (
              <TransactionRow transaction={transaction} key={transaction._id} readOnly={readOnly} />
            ))}
        </TableItems>
      </TransactionTable>
    )
  }
}
 
export default TransactionDisplay
