// @flow
import React, { Component } from 'react';
import PropTypes from 'prop-types';
 
type Props = {
  handleSubmit: Function
};
 
type State = {
  amountInputValue: string
};
 
class ItemForm extends Component<Props, State> {
  static defaultProps = {
    handleSubmit: undefined,
  };
 
  state = {
    amountInputValue: '',
  };
  handleChange = ev => {
    this.setState({ amountInputValue: `$${ev.target.value}` });
  };
  render() {
    const now = new Date(Date.now());
    const resDate = now.toISOString().slice(0, 10);
    return (
      <div data-testid="item-form">
        <form
          onSubmit={ev => {
            ev.preventDefault();
            const { itemName, amount, date, person } = ev.target.elements;
            this.props.handleSubmit({
              itemName: itemName.value,
              amount: amount.value,
              date: date.valueAsNumber,
              person: person.value,
            });
          }}
        >
          <h4>Enter your purchase</h4>
          <label htmlFor="itemName">Item Name</label>
          <input type="text" id="itemName" aria-label="itemName" placeholder="Ramen Noodles" />
          <label htmlFor="amount">Item Amount</label>
          <input
            type="string"
            id="amount"
            aria-label="amount"
            value={this.state.amountInputValue}
            placeholder="$3.58"
            onChange={this.handleChange}
          />
          <label htmlFor="date">Date of Purchase</label>
          <input type="date" id="date" aria-label="date" defaultValue={resDate} />
          <label htmlFor="person">Purchaser</label>
          <input type="text" id="person" aria-label="person" placeholder="Brianna" />
          <button type="submit">Submit</button>
        </form>
      </div>
    );
  }
}
 
export default ItemForm;
